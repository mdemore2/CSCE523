
(operator
  MOVE-WEST
  (params (<car> Car) (<noseLoc> Space) (<tailLoc> Space) (<dest> Space))
  (preconds 
	(empty <dest>) (nose <car> <noseLoc>) (tail <car> <tailLoc>) (east <tailLoc> <dest>) (east <noseLoc> <tailLoc>)
  )
  (effects 
	(empty <noseLoc>) (nose <car> <tailLoc>) (tail <car> <dest>)
  )
)
